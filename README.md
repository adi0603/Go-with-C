long power(int base, int exponent)
{
    int i =0;
    long result = 1;
    while ( ++i <= exponent )
    {
        result *= base;
    }
    return result;
}

unsigned long compute_sum(int terms)
{
    int i = 1, base = 1;
    unsigned long sum = 0;
    while ( i++ <= terms )
    {
        sum += power(base, base);
        printf("%ld ", sum);
        base++;
    }
    return sum;
